{% comment %} Series navigation that orders files alphabetically by filename (GitHub style). Includes all .md and .html files in the series. Displays "Page X of Y" and previous/next links. Works with Jekyll collections OR regular site.pages directories. Override series_dir by passing e.g. {% include series-nav.html series_dir="/my/path/" %} {% endcomment %} {% comment %} Determine the series base directory (only used for regular pages) {% endcomment %} {% if include.series_dir %} {% assign series_base = include.series_dir %} {% else %} {% assign series_base = page.dir %} {% endif %} {% comment %} Ensure trailing slash for series_base {% endcomment %} {% assign last_char = series_base | slice: -1, 1 %} {% unless last_char == '/' %} {% assign series_base = series_base | append: '/' %} {% endunless %} {% comment %} Build the list of pages in the same series {% endcomment %} {% assign series_pages = nil %} {% comment %} Case 1: Page belongs to a Jekyll collection {% endcomment %} {% if page.collection %} {% assign series_pages = site[page.collection] | sort: "name" %} {% comment %} Case 2: Regular page – use all .md/.html files under series_base {% endcomment %} {% else %} {% assign all_pages = site.pages | where_exp: "p", "p.path contains series_base" %} {% assign temp_pages = "" | split: "" %} {% for p in all_pages %} {% assign ext = p.name | split: "." | last %} {% if ext == "md" or ext == "html" %} {% assign temp_pages = temp_pages | push: p %} {% endif %} {% endfor %} {% assign series_pages = temp_pages | sort: "name" %} {% endif %} {% comment %} Find current page index {% endcomment %} {% assign current_index = nil %} {% for p in series_pages %} {% if p.path == page.path %} {% assign current_index = forloop.index0 %} {% assign total = series_pages.size %} {% assign prev_index = current_index | minus: 1 %} {% assign next_index = current_index | plus: 1 %} {% break %} {% endif %} {% endfor %} {% comment %} Show navigation only if more than one page {% endcomment %} {% if total > 1 and current_index != nil %}
