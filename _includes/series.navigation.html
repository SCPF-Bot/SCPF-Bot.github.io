{%- comment %} Usage example: {% include series_navigation.html series_dir="/my-series" %} If series_dir is omitted, it defaults to the current page's directory. The component sorts pages by their file path (alphabetical). You can override sorting by adding an `order` front matter variable. {% endcomment -%} {%- assign series_dir = include.series_dir | default: page.dir -%} {%- unless series_dir ends_with '/' %}{% assign series_dir = series_dir | append: '/' %}{% endunless -%} {%- comment %} Gather all pages under series_dir, filter by extension, and sort {% endcomment -%} {%- assign all_series_pages = site.pages | where_exp: "p", "p.url contains series_dir" | where_exp: "p", "p.url != series_dir" # exclude the directory itself if it's a page | where_exp: "p", "p.path matches '\\.(md|html)$'" | sort: "path" -%} {%- comment %} Optional: resort by front matter `order` if needed {% endcomment -%} {%- assign sorted_pages = "" | split: "" -%} {%- assign unordered = all_series_pages -%} {%- for i in (0..unordered.size) -%} {%- assign smallest = nil -%} {%- assign smallest_index = nil -%} {%- for p in unordered -%} {%- assign order = p.order | default: 999999 -%} {%- if smallest == nil or order < smallest -%} {%- assign smallest = order -%} {%- assign smallest_page = p -%} {%- assign smallest_index = forloop.index0 -%} {%- endif -%} {%- endfor -%} {%- if smallest_page -%} {%- assign sorted_pages = sorted_pages | push: smallest_page -%} {%- assign unordered = unordered | pop: smallest_index -%} {%- endif -%} {%- endfor -%} {%- if sorted_pages.size == 0 -%}{%- assign sorted_pages = all_series_pages -%}{%- endif -%} {%- comment %} Find current page index {% endcomment -%} {%- assign current_index = nil -%} {%- for p in sorted_pages -%} {%- if p.url == page.url -%} {%- assign current_index = forloop.index0 -%} {%- break -%} {%- endif -%} {%- endfor -%} {%- assign total_pages = sorted_pages.size -%} {%- if total_pages > 0 -%}
{%- comment %} Previous button (disabled if first page) {% endcomment -%} {%- if current_index > 0 -%} {%- assign prev_page = sorted_pages[current_index | minus: 1] -%} Previous: {{ prev_page.title | default: prev_page.name }} {%- else -%} Previous {%- endif -%} {%- comment %} Page counter {% endcomment -%} Page {{ current_index | plus: 1 }} of {{ total_pages }} {%- comment %} Next button (disabled if last page) {% endcomment -%} {%- if current_index < total_pages | minus: 1 -%} {%- assign next_page = sorted_pages[current_index | plus: 1] -%} Next: {{ next_page.title | default: next_page.name }} {%- else -%} Next {%- endif -%}
