{%- comment %} Simple series navigation for sibling pages in the same directory. Only .md and .html files are considered. Sorted alphabetically by filename. Disabled buttons keep layout consistent. FontAwesome icons used for arrows. {% endcomment -%} {%- assign current_dir = page.dir -%} {%- assign siblings = site.pages | where: "dir", current_dir | where_exp: "p", "p.path matches '\\.(md|html)$'" | sort: "name" -%} {%- if siblings.size > 0 -%} {%- assign current_index = nil -%} {%- for sibling in siblings -%} {%- if sibling.url == page.url -%} {%- assign current_index = forloop.index0 -%} {%- break -%} {%- endif -%} {%- endfor -%}
{%- comment %} Previous button {% endcomment -%} {%- if current_index > 0 -%} {%- assign prev = siblings[current_index | minus: 1] -%} Previous: {{ prev.title | default: prev.name }} {%- else -%} Previous {%- endif -%} {%- comment %} Page counter {% endcomment -%} Page {{ current_index | plus: 1 }} of {{ siblings.size }} {%- comment %} Next button {% endcomment -%} {%- if current_index < siblings.size | minus: 1 -%} {%- assign next = siblings[current_index | plus: 1] -%} Next: {{ next.title | default: next.name }} {%- else -%} Next {%- endif -%}
{%- endif -%}
